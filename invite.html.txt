  <script>
    // Простий JS для відкриття/закриття модалки та клієнтської валідації форми
    (function(){
      const openBtn = document.getElementById('rsvp-open');
      const modal = document.getElementById('rsvp-modal');
      const closeBtn = document.getElementById('rsvp-close');
      const cancelBtn = document.getElementById('cancel-rsvp');
      const form = document.getElementById('rsvp-form');
      const thanks = document.getElementById('rsvp-thanks');

      function openModal(){
        modal.style.display = 'flex';
        document.body.style.overflow = 'hidden';
        // ставимо фокус у перше поле
        setTimeout(()=> document.getElementById('fullName').focus(), 100);
      }
      function closeModal(){
        modal.style.display = 'none';
        document.body.style.
overflow = '';
        // очищуємо стан форми після закриття, якщо показане віддякування — приховати
        if(thanks.style.display === 'block'){
          thanks.style.display = 'none';
          form.style.display = 'block';
          form.reset();
        }
      }

      openBtn.addEventListener('click', openModal);
      closeBtn.addEventListener('click', closeModal);
      cancelBtn.addEventListener('click', closeModal);

      // Закрити модалку при ESC
      window.addEventListener('keydown', function(e){
        if(e.key === 'Escape' && modal.style.display === 'flex') closeModal();
      });

      // Клік поза модалкою закриває
      modal.addEventListener('click', function(e){
        if(e.target === modal) closeModal();
      });

      // Обробка відправки (клієнтська)
      form.addEventListener('submit', function(e){
        e.preventDefault();
        // Проста валідація: перевірка на значення для обов'язкових
        const fullName = document.getElementById('fullName').value.trim();
        const contact = document.getElementById('email').value.trim();
        const attendance = document.getElementById('attendance').value;

        if(!fullName || !contact){
          alert('Будь ласка, заповніть ПІБ та контактні дані.');
          return;
        }

        // Імітація відправки — показати повідомлення "дякуємо"
        form.style.display = 'none';
        thanks.style.display = 'block';

        // Тут: якщо буде бекенд, можна зробити fetch() на API.
        // Поки — імітація асинхронної відповіді.
        setTimeout(function(){
          // Після 3 секунд залишимо модалку відкритою з повідомленням.
        }, 3000);
      });
    })();
  </script>
